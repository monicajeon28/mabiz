// 방법 1: HTML 엔티티 인코딩 (안전)
function encodeHtml(str: string) {
  const map: Record<string, string> = {
    '&': '&amp;',
    '<': '&lt;',
    '>': '&gt;',
    '"': '&quot;',
    "'": '&#039;'
  };
  return str.replace(/[&<>"']/g, (char) => map[char]);
}

const imgTags = images.map((img, idx) => {
  const src = `https://lh3.googleusercontent.com/d/${encodeHtml(img.driveFileId)}=w1200`;
  const ar = img.width && img.height ? `aspect-ratio:${img.width}/${img.height};` : "";
  // 동적 alt text
  const altText = img.altText || `이미지 ${idx + 1}/${images.length}`;
  return `<img src="${src}" alt="${encodeHtml(altText)}" style="width:100%;display:block;${ar}" loading="lazy" />`;
}).join("\n");

// 방법 2: 폼 필드 동적화 (설정 반영)

### 🔴 Issue 2: Race Condition (uploadFiles)
**파일:** `src/app/(dashboard)/landing-pages/[id]/page.tsx:318-336`

**문제:**
```typescript
// 현재 (위험)
const uploadFiles = async (files: FileList) => {
  setUploading(true); setError("");
  for (let i = 0; i < files.length; i++) {
    const file = files[i];
    if (!file.type.startsWith("image/")) continue;
    if (file.size > 20 * 1024 * 1024) { setError(`...`); continue; }
    const formData = new FormData();
    formData.append("file", file);
    formData.append("landingPageId", id);
    formData.append("sortOrder", String(images.length + i));  // ← 문제!
    try {
      const res = await fetch("/api/landing-pages/images", { method: "POST", body: formData });
      const data = await res.json();  // ← 에러 미처리
      if (data.ok) setImages((prev) => [...prev, data.image]);
      else setError(data.message ?? `...`);
    } catch { setError(`...`); }
  }
  setUploading(false);  // ← 루프 전체 후에만 실행
};
```

**위험점:**
1. **sortOrder 계산 오류**: `images.length + i`는 루프 중 변할 수 있음
   - 파일 5개 동시 업로드 시 sortOrder가 [5, 6, 7, 8, 9]가 아니라 [5, 5, 5, 5, 5]가 될 수 있음
2. **상태 불일치**: setUploading(false)가 루프 후 한 번만 실행
   - 네트워크 느린 경우: 첫 파일 로드 중 다음 파일 업로드 시작 → 순서 꼬임
3. **에러 누적**: setError()가 마지막 에러만 표시

**영향:**
- 🚨 DB sortOrder 중복 or 순서 꼬임
- 🚨 UI 진행률 표시 불가
- 🚨 부분 실패 감지 불가

**해결 방법:**
```typescript
interface UploadResult {
  success: Array<{ fileName: string; image: UploadedImage }>;
  failed: Array<{ fileName: string; reason: string }>;
}

const uploadFiles = async (files: FileList) => {
  setUploading(true);
  setError("");
  const results: UploadResult = { success: [], failed: [] };
  
  const validFiles = Array.from(files).filter(file => {
    if (!file.type.startsWith("image/")) {
      results.failed.push({ fileName: file.name, reason: "이미지 파일만 가능" });
      return false;
    }
    if (file.size > 20 * 1024 * 1024) {
      results.failed.push({ fileName: file.name, reason: "20MB 초과" });
      return false;
    }
    return true;
  });

  // 배치 처리 (3~5개씩)
  const BATCH_SIZE = 3;
  for (let i = 0; i < validFiles.length; i += BATCH_SIZE) {
    const batch = validFiles.slice(i, i + BATCH_SIZE);
    
    // 배치 내 파일들을 병렬 처리
    const batchResults = await Promise.allSettled(
      batch.map(file => uploadSingleFile(file, images.length + i + batch.indexOf(file)))
    );

    // 결과 수집
    batchResults.forEach((result, idx) => {
      const file = batch[idx];
      if (result.status === "fulfilled") {
        const { ok, image, message } = result.value;
        if (ok) {
          results.success.push({ fileName: file.name, image });
          setImages((prev) => [...prev, image]);
        } else {
          results.failed.push({ fileName: file.name, reason: message || "업로드 실패" });
        }
      } else {
        results.failed.push({ fileName: file.name, reason: result.reason?.message || "네트워크 오류" });
      }
    });

    // 진행률 업데이트
    setUploadProgress({
      processed: i + batch.length,
      total: validFiles.length,
      percent: Math.round(((i + batch.length) / validFiles.length) * 100),
    });
  }

  // 결과 표시
  if (results.success.length > 0) {
    setSuccess(`✅ ${results.success.length}개 업로드 성공`);
  }
  if (results.failed.length > 0) {
    setError(`❌ ${results.failed.length}개 실패:\n${results.failed.map(f => `- ${f.fileName}: ${f.reason}`).join("\n")}`);
  }

  setUploading(false);
};

// 단일 파일 업로드
async function uploadSingleFile(file: File, sortOrder: number) {
  const formData = new FormData();
  formData.append("file", file);
  formData.append("landingPageId", id);
  formData.append("sortOrder", String(sortOrder));
  
  const res = await fetch("/api/landing-pages/images", { 
    method: "POST", 
    body: formData 
  });
  
  if (!res.ok) {
    throw new Error(`HTTP ${res.status}`);
  }
  
  return await res.json();
}
```

const handleDragEnd = async () => {
