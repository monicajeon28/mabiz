const doAction = async (id: number, action: "approve" | "reject" | "refund") => {

function validateInquiryForm(data: Record<string, any>): { valid: boolean; errors: Record<string, string> } {
  const errors: Record<string, string> = {};
  
  if (!data.memberName || data.memberName.trim().length < 2) {
    errors.memberName = '이름은 2자 이상이어야 합니다';
  }
  if (!data.phone || !/^01[0-9]-?\d{3,4}-?\d{4}$/.test(data.phone)) {
    errors.phone = '올바른 휴대폰 번호를 입력하세요';
  }
  if (!data.content || data.content.trim().length < 10) {
    errors.content = '문의 내용은 10자 이상이어야 합니다';
  }
  
  return {
    valid: Object.keys(errors).length === 0,
    errors,
  };
}

// 2. 서버 (보안)
export async function POST(req: Request) {
  try {
