### 7️⃣ **DB 최적화 렌즈** ⚠️ P1
**새 테이블:**
```prisma
model CrmMarketingCampaign {
  id              String @id @default(cuid())
  organizationId  String
  groupId         String       # 고객 그룹
  title           String       # 캠페인명 (예: "5월 VIP 고객 재구매")
  
  # 메시지 선택
  sendEmail       Boolean @default(false)
  sendSms         Boolean @default(true)
  includeLanding  Boolean @default(true)
  
  # 스케줄
  sendAt          DateTime
  repeatRule      String?      # CRON (예: "0 9 * * 1" = 매주 월 9시)
  
  # 통계
  totalCount      Int @default(0)      # 발송 대상 수
  sentCount       Int @default(0)      # 발송 완료
  openedCount     Int @default(0)      # 열람
  clickedCount    Int @default(0)      # 클릭
  registeredCount Int @default(0)      # 신청

  group           ContactGroup @relation(fields: [groupId], references: [id])
  messages        CrmMarketingMessage[]
  
  createdAt       DateTime @default(now())
  updatedAt       DateTime @updatedAt
  
  @@index([organizationId, groupId])
  @@index([organizationId, sendAt])
}

model CrmMarketingMessage {
  id              String @id @default(cuid())
  campaignId      String
  recipientId     String       # 고객 ID
  
  # 전송 상태
  emailSent       Boolean @default(false)
  smsSent         Boolean @default(false)
  emailOpenedAt   DateTime?
  linkClickedAt   DateTime?
  registeredAt    DateTime?
  
  # 결과
  registrationId  String?      # CrmLandingRegistration 링크
  
  campaign        CrmMarketingCampaign @relation(fields: [campaignId], references: [id], onDelete: Cascade)
  
  createdAt       DateTime @default(now())
  
  @@unique([campaignId, recipientId])
  @@index([campaignId])
  @@index([recipientId])
}
```

```typescript
// POST /api/marketing/campaigns/[id]/send
async function sendCampaign(campaignId: string) {
  const campaign = await getCampaign(campaignId);
  const group = await getGroupMembers(campaign.groupId);
  
  // 배치 처리: 100명씩
  const batches = chunk(group, 100);
  
  for (const batch of batches) {
    await Promise.all(batch.map(async (recipient) => {
      if (campaign.sendEmail) 
        await sendEmail(recipient, campaign);
      if (campaign.sendSms)
        await sendSms(recipient, campaign);
      if (campaign.includeLanding)
        await appendLandingLink(recipient, campaign);
    }));
  }
}
```
