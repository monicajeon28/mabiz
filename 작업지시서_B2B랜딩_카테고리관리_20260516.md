```prisma
model B2BLandingPage {
  // 기존 필드들...
  id              String @id @default(cuid())
  organizationId  String
  partnerId       String?
  title           String
  slug            String
  // ... 기타 필드

  @@unique([organizationId, slug])
  @@index([organizationId, category])
  @@index([organizationId, isActive])
}

// ✨ NEW: 카테고리 메타데이터 테이블
model B2BLandingCategory {
  id              String @id @default(cuid())
  organizationId  String
  name            String
  description     String?
  color           String? @default("#3b82f6")  // 카테고리별 색상 (향후)

  createdAt       DateTime @default(now())
  updatedAt       DateTime @updatedAt

  @@unique([organizationId, name])
  @@index([organizationId])
}
```

#### **UI 1: 페이지 편집 화면에 카테고리/상태 추가**

**파일:** `src/app/(dashboard)/b2b-editor/[id]/page.tsx` (수정)

```typescript
// 1. 상태 추가
const [category, setCategory] = useState("");
const [isActive, setIsActive] = useState(true);
const [categories, setCategories] = useState<string[]>([]);

// 2. 카테고리 목록 로드
useEffect(() => {
  fetch('/api/b2b-landing/categories')
    .then(r => r.json())
    .then(d => setCategories(d.data.categories.map(c => c.name)));
}, []);

// 3. UI 렌더링
return (
  <div className="space-y-4">
    {/* 기존 필드들 */}
    <input value={title} onChange={...} placeholder="제목" />
    
    {/* ✨ NEW: 카테고리 */}
    <div>
      <label className="block text-sm font-medium mb-1">카테고리</label>
      <div className="flex gap-2">
        <select 
          value={category} 
          onChange={(e) => setCategory(e.target.value)}
          className="flex-1 border rounded px-3 py-2"
        >
          <option value="">선택 안 함</option>
          {categories.map(cat => (
            <option key={cat} value={cat}>{cat}</option>
          ))}
        </select>
        <button 
          onClick={() => {
            const newCat = prompt("새 카테고리 이름:");
            if (newCat) {
              fetch('/api/b2b-landing/categories', {
                method: 'POST',
                body: JSON.stringify({ name: newCat })
              }).then(() => {
                // 카테고리 목록 새로고침
                setCategories([...categories, newCat]);
              });
            }
          }}
          className="px-3 py-2 bg-blue-500 text-white rounded"
        >
          + 새로
        </button>
      </div>
    </div>

    {/* ✨ NEW: 활성/비활성 토글 */}
    <div>
      <label className="block text-sm font-medium mb-1">상태</label>
      <div className="flex items-center gap-2">
        <input 
          type="checkbox" 
          checked={isActive}
          onChange={(e) => setIsActive(e.target.checked)}
          className="w-4 h-4"
        />
        <span>{isActive ? "✅ 활성 (보여짐)" : "❌ 비활성 (숨김)"}</span>
      </div>
    </div>

    {/* 저장 버튼 */}
    <button 
      onClick={savePage}
```

**CategoryManagementTab 컴포넌트:**

```typescript
function CategoryManagementTab() {
  const [categories, setCategories] = useState([]);
  const [newCategoryName, setNewCategoryName] = useState('');

  // 1. 카테고리 목록 로드
  useEffect(() => {
    fetch('/api/b2b-landing/categories')
      .then(r => r.json())
      .then(d => setCategories(d.data.categories));
  }, []);

  return (
    <div className="space-y-4">
      {/* 새 카테고리 추가 */}
      <div className="flex gap-2">
        <input
          type="text"
          value={newCategoryName}
          onChange={(e) => setNewCategoryName(e.target.value)}
          placeholder="새 카테고리 이름 입력"
          className="flex-1 border rounded px-3 py-2"
        />
        <button
          onClick={() => {
            fetch('/api/b2b-landing/categories', {
              method: 'POST',
              body: JSON.stringify({ name: newCategoryName })
            }).then(() => {
              setNewCategoryName('');
              // 목록 새로고침
            });
          }}
          className="px-4 py-2 bg-blue-500 text-white rounded"
        >
          + 추가
        </button>
      </div>

      {/* 카테고리 목록 */}
      <div className="space-y-2">
        {categories.map(cat => (
          <div
            key={cat.id}
            className="flex items-center justify-between bg-white border rounded p-4"
          >
            <div>
              <h3 className="font-medium">{cat.name}</h3>
              <p className="text-sm text-gray-500">
                {cat.usageCount}개 페이지에서 사용 중
              </p>
            </div>
            <button
              onClick={() => {
                if (cat.usageCount > 0) {
                  alert(`⚠️ ${cat.usageCount}개 페이지에서 사용 중입니다.\n먼저 카테고리를 변경해주세요.`);
                } else {
                  // 삭제 로직
                }
              }}
              className="px-3 py-1 text-red-500 border border-red-500 rounded hover:bg-red-50"
            >
              삭제
            </button>
          </div>
        ))}
      </div>

      {/* 설명 */}
      <div className="bg-blue-50 border border-blue-200 rounded p-4 text-sm text-blue-900">
        💡 팁: 카테고리는 페이지 편집 화면에서도 빠르게 선택/추가할 수 있습니다.
      </div>
    </div>
  );
}
```
