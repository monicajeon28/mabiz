## 📋 해결 방안 (3단계)

### Phase 1: 로컬 이미지 다운로드 수정

**문제**: `driveFileId`가 없음 → 다운로드 API 호출 안 됨

**해결책**: 로컬 이미지는 새로운 다운로드 엔드포인트 필요

#### Step 1-1: 새 다운로드 API 생성

파일: `src/app/api/image-library/[id]/download/route.ts` **(새 파일)**

```typescript
import { NextResponse } from 'next/server';
import { getAuthContext } from '@/lib/rbac';
import prisma from '@/lib/prisma';
import sharp from 'sharp';
import path from 'path';
import fs from 'fs';

/**
 * GET /api/image-library/[id]/download
 * 로컬 이미지를 다운로드하고 워터마크를 합성하여 PNG로 반환.
 */
export async function GET(
  _req: Request,
  { params }: { params: { id: string } }
) {
  try {
    const ctx = await getAuthContext();
    const imageId = params.id;

    // DB에서 이미지 조회
    const image = await prisma.imageAsset.findUnique({
      where: { id: imageId },
      select: { id: true, title: true, imagePath: true, organizationId: true },
    });

    if (!image || image.organizationId !== ctx.organizationId) {
      return NextResponse.json(
        { ok: false, error: 'Image not found' },
        { status: 404 }
      );
    }

    // 파일 시스템에서 읽기
    const filePath = path.join(process.cwd(), 'public', image.imagePath);
    if (!fs.existsSync(filePath)) {
      return NextResponse.json(
        { ok: false, error: 'File not found' },
        { status: 404 }
      );
    }

    const buffer = fs.readFileSync(filePath);

    // Sharp 워터마크 합성 (다운로드 API와 동일)
    const meta = await sharp(buffer).metadata();
    const w = meta.width ?? 800;
    const h = meta.height ?? 600;

    // 회색 오버레이 + 로고
    const overlay = await sharp({
      create: { width: w, height: h, channels: 4, background: { r: 0, g: 0, b: 0, alpha: 0.3 } },
    }).png().toBuffer();

    const logoPath = path.join(process.cwd(), 'public', 'logo.png');
    let watermarkInput: Buffer;

    if (fs.existsSync(logoPath)) {
      const logoSize = Math.max(80, Math.floor(w * 0.3));
      watermarkInput = await sharp(logoPath)
        .resize({ width: logoSize, withoutEnlargement: true })
        .png()
        .toBuffer();
    } else {
      // Fallback: 텍스트 워터마크
      const fontSize = Math.max(24, Math.min(80, Math.floor(w / 10)));
      watermarkInput = Buffer.from(
        `<svg width="${w}" height="${h}" xmlns="http://www.w3.org/2000/svg">
          <text x="50%" y="50%" text-anchor="middle" dominant-baseline="middle"
            font-family="Arial" font-size="${fontSize}" font-weight="bold"
            fill="white" opacity="0.6"
            transform="rotate(-30,${w / 2},${h / 2})">CRUISEDOT</text>
        </svg>`
      );
    }

    const result = await sharp(buffer)
      .composite([
        { input: overlay, blend: 'over' },
        { input: watermarkInput, gravity: 'center', blend: 'over' },
      ])
      .png()
      .toBuffer();

    const safeName = encodeURIComponent(image.title.replace(/\.[^.]+$/, ''));
    return new Response(result as unknown as BodyInit, {
      headers: {
        'Content-Type': 'image/png',
        'Content-Disposition': `attachment; filename="${safeName}.png"; filename*=UTF-8''${safeName}.png`,
        'Content-Length': String(result.length),
        'Cache-Control': 'no-store',
      },
    });
  } catch (err) {
    const msg = err instanceof Error ? err.message : String(err);
    return NextResponse.json(
      { ok: false, error: msg },
      { status: 500 }
    );
  }
}
```

// 변경 후
const downloadLocalAsset = (asset: ImageAsset) => {
  // Google Drive 이미지
  if (asset.driveFileId) {
    const url = `/api/image-library/download?id=${asset.driveFileId}&name=${encodeURIComponent(asset.title)}`;
    const a = document.createElement('a');
    a.href = url;
    a.download = asset.title.replace(/\.[^.]+$/, '') + '_watermark.png';
    a.click();
  }
  // 로컬 이미지
  else {
    const url = `/api/image-library/${asset.id}/download`;
    const a = document.createElement('a');
    a.href = url;
    a.download = asset.title.replace(/\.[^.]+$/, '') + '_watermark.png';
    a.click();
  }
};
