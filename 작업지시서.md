export function validateContactGroupInput(body: any) {
  const errors: string[] = [];

  if (!body.name || typeof body.name !== 'string') {
    errors.push('name은 필수입니다');
  } else if (body.name.trim().length < 1 || body.name.length > 100) {
    errors.push('name은 1~100자여야 합니다');

  if (!body.color || typeof body.color !== 'string') {
    errors.push('color는 필수입니다');
  } else if (!/^#[0-9a-fA-F]{6}$/.test(body.color)) {

    errors.push('funnelId는 문자열이어야 합니다');
  }

  if (body.description && body.description.length > 500) {
    errors.push('description은 500자 이하여야 합니다');
  }

  return { valid: errors.length === 0, errors };
}

// ✅ route.ts에서 사용

function parseIntSafe(str: string, defaultValue: number): number {

  return Number.isNaN(parsed) ? defaultValue : Math.max(1, parsed);
}

// ✅ src/lib/timeout.ts
export function withTimeout<T>(promise: Promise<T>, ms: number, label: string): Promise<T> {
  return Promise.race([
    promise,
    new Promise<T>((_, reject) =>
      setTimeout(() => reject(new Error(`TIMEOUT: ${label} after ${ms}ms`)), ms)
    )
  ]);
}

// 사용
const [prospects, count, statuses] = await Promise.all([
  withTimeout(
    prisma.crmB2BProspect.findMany({ ... }),
    5000,
    'B2B prospects list'
  ),
  withTimeout(

  ),
